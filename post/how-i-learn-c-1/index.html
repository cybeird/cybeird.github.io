#include <cstdio>
#include <cstring>
#include <iostream>
#include <algorithm>
using namespace std;

class Time
{
	private:
		int hour;
		int minute;
		int sec;
	public:
		void getin(void)
		{
			cin >> hour;
			cin >> minute;
			cin >> sec;
		}
		void out(void)
		{
			cout << hour << ":" << minute << ":" << sec <<endl;
		}
};
int main()
{
	Time t1;
	t1.getin();
	t1.out();
	//int sundea;cin>>sundea;
	return 0;
}
