#include<stdio.h>
int lowbit(int x)
{
  return x & (-x);
}
int bit[10001];
int a[100001];
int n,l,r;
int built()
{
  for(int i=1;i<=n;i++)
   {
   	bit[i]=a[i];
   	for(int j=i-1;j>i-lowbit(i);j-=lowbit(j))
   	 bit[i]+=bit[j];
   }
}
int sum(int k)
{
  int ans=0;
  for(int i=k;i>0;i-=lowbit(i))
   ans+=bit[i];
  return ans; 
}
int change(int x,int delta)
{
  for(int i=x;i<=n;i+=lowbit(i))
   bit[i]+=delta;
}
int x,delta;
int main()
{
  scanf("%d",&n);
  for(int i=1;i<=n;i++)
   scanf("%d",&a[i]);
  built();
  scanf("%d%d",&l,&r); 
  printf("%d\n",sum(r)-sum(l-1));
  scanf("%d%d",&x,&delta);
  change(x,delta);
  scanf("%d%d",&l,&r);
  printf("%d\n",sum(r)-sum(l-1));
}
